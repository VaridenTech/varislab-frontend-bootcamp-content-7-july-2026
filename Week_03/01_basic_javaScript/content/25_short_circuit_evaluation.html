การประเมินผลแบบลัดวงจร

การประเมินผลแบบลัดวงจรเป็นคุณสมบัติที่ทรงพลังของตัวดำเนินการทางตรรกะใน JavaScript (&& และ ||) ซึ่งช่วยให้โค้ดมีประสิทธิภาพและกระชับขึ้น

การประเมินผลแบบลัดวงจรคืออะไร?

หมายถึง JavaScript จะหยุดประเมินนิพจน์ทางตรรกะทันทีที่ทราบผลลัพธ์สุดท้าย โดยจะ “ลัดวงจร” หรือข้ามส่วนที่เหลือของนิพจน์

ตัวดำเนินการ && (AND)

สำหรับ AND operand ทั้งสอง ต้องเป็น true ผลลัพธ์จึงเป็น true หาก operand แรกเป็น false JavaScript จะรู้ว่าทั้งนิพจน์เป็น false จึงไม่ประเมิน operand ที่สอง

const a = false
const b = true

// JavaScript ตรวจเฉพาะ 'a' พบว่าเป็น false แล้วหยุด
const result = a && b
console.log(result) // false

// หากมีการเรียกฟังก์ชันที่ใช้ทรัพยากรมาก:
function expensiveFunction() {
  console.log("This is an expensive operation!")
  return true
}

// expensiveFunction() จะไม่ถูกเรียก เพราะ a เป็น false
const result2 = a && expensiveFunction()
console.log(result2) // false (และไม่มี console.log จาก expensiveFunction)

สิ่งนี้ทำให้ && เหมาะสำหรับการเข้าถึงพร็อพเพอร์ตีแบบมีเงื่อนไข:

// เข้าถึงพร็อพเพอร์ตีอย่างปลอดภัย
const user = null
user && user.name && console.log(user.name) // ไม่เกิดข้อผิดพลาด

ตัวดำเนินการ || (OR)

สำหรับ OR หาก operand แรกเป็น true JavaScript จะรู้ว่าทั้งนิพจน์เป็น true จึงไม่ประเมิน operand ที่สอง

const a = true
const b = false

// JavaScript ตรวจเฉพาะ 'a' พบว่าเป็น true แล้วหยุด
const result = a || b
console.log(result) // true

function expensiveFunction() {
  console.log("This expensive function runs!")
  return false
}

// expensiveFunction() จะไม่ถูกเรียก เพราะ a เป็น true
const result2 = a || expensiveFunction()
console.log(result2) // true (และไม่มี console.log จาก expensiveFunction)

สิ่งนี้ทำให้ || เหมาะสำหรับกำหนดค่าสำรอง:

const userInput = ""
const defaultName = "Guest"

// หาก userInput เป็น falsy ให้ใช้ defaultName
const displayName = userInput || defaultName
console.log(displayName) // "Guest"

// ตัวอย่างเพิ่มเติม
const config = {
  theme: null,
  timeout: 0,
}

const theme = config.theme || "light" // "light"
const timeout = config.timeout || 5000 // 5000
const debug = config.debug || false // false

ค่าที่ส่งกลับในการประเมินผลแบบลัดวงจร

สำคัญ: ตัวดำเนินการแบบลัดวงจรไม่ได้ส่งกลับเพียง true หรือ false แต่ส่งค่าจริงที่กำหนดผลลัพธ์กลับมา!

AND && ส่งกลับ:

// ส่งค่า falsy แรก หรือค่าสุดท้ายหากทุกค่าเป็น truthy
console.log(false && "hello") // false
console.log(true && "hello") // "hello"
console.log("hi" && "bye") // "bye"
console.log("" && "hello") // "" (String ว่างเป็น falsy)
console.log(5 && 10 && 15) // 15 (ทุกค่าเป็น truthy จึงส่งค่าสุดท้าย)
console.log(5 && 0 && 15) // 0 (ค่า falsy แรก)

OR || ส่งกลับ:

// ส่งค่า truthy แรก หรือค่าสุดท้ายหากทุกค่าเป็น falsy
console.log(false || "hello") // "hello"
console.log(true || "hello") // true
console.log("hi" || "bye") // "hi"
console.log("" || "hello") // "hello"
console.log(null || undefined || "default") // "default"
console.log(0 || false || null) // null (ทุกค่าเป็น falsy จึงส่งค่าสุดท้าย)

ทางเลือกสมัยใหม่

JavaScript มีตัวดำเนินการใหม่สองตัวที่ใช้แทนรูปแบบลัดวงจรทั่วไปบางแบบได้:

Nullish Coalescing ??

ตัวดำเนินการ nullish coalescing ถือว่าเฉพาะ null และ undefined เป็น falsy จึงเหมาะกับค่าเริ่มต้นเมื่อ 0 หรือ false เป็นค่าที่ใช้ได้:

const config = {
  timeout: 0, // เราต้องการเก็บค่า 0 นี้ไว้!
  debug: null, // ค่านี้ควรได้รับค่าเริ่มต้น
}

// ใช้ || (ถือว่า 0 เป็น falsy)
const timeout1 = config.timeout || 5000 // 5000 (❌)
const debug = config.debug || true // true (✅)

// ใช้ ?? (ถือว่าเฉพาะ null/undefined เป็น falsy)
const timeout2 = config.timeout ?? 5000 // 0 (✅)
const debug = config.debug ?? true // true (✅)

Optional Chaining ?.

Optional chaining จะเข้าถึงพร็อพเพอร์ตีเฉพาะเมื่อค่าก่อนหน้าไม่ใช่ null หรือ undefined เหมาะสำหรับเข้าถึงพร็อพเพอร์ตีที่ซ้อนลึกอย่างปลอดภัย:

const user = {
  profile: {
    settings: {
      theme: "dark",
    },
  },
}

// วิธีเดิมด้วย &&
const theme1 =
  user && user.profile && user.profile.settings && user.profile.settings.theme

// วิธีใหม่ด้วย optional chaining
const theme2 = user?.profile?.settings?.theme

// ใช้กับเมธอดได้เช่นกัน
user?.notify?.()
