บทนำเกี่ยวกับฟังก์ชัน

ฟังก์ชันเก็บตรรกะไว้ใช้ในภายหลังในโค้ด คล้ายกับที่ตัวแปรเก็บค่าไว้ใช้ในภายหลัง

ทำไมเราจึงต้องใช้ฟังก์ชัน?

ฟังก์ชันช่วยให้เราไม่ต้องเขียนโค้ดเดิมหลายครั้ง มาดูตัวอย่างกัน:

console.log("Hello")

// ... มีโค้ดอื่นอีกมากมายอยู่ตรงนี้ ...

console.log("Hello") // เรากำลังเขียนสิ่งเดิมซ้ำ!

หากเราต้องการเปลี่ยน “Hello” เป็น “Hi” เราจะต้องจำให้ได้ว่าต้องแก้ไขทุกจุด นี่คือจุดที่ฟังก์ชันมีประโยชน์

การสร้างฟังก์ชันแรกของคุณ

นี่คือวิธีเขียนฟังก์ชันใน JavaScript:

function sayHi() {
  console.log("Hi")
}

เรามาแยกดูแต่ละส่วนกัน:

  1. function - คีย์เวิร์ดที่บอก JavaScript ว่าเรากำลังสร้างฟังก์ชัน
  2. sayHi - ชื่อฟังก์ชันของเรา (ใช้กฎการตั้งชื่อเดียวกับตัวแปร)
  3. () - วงเล็บสำหรับพารามิเตอร์ (ตอนนี้ยังว่างอยู่)
  4. {} - วงเล็บปีกกาที่ครอบโค้ดของฟังก์ชัน

การตั้งชื่อฟังก์ชัน

ชื่อฟังก์ชันใช้กฎเดียวกับตัวแปร:

  • เริ่มต้นด้วยตัวอักษรพิมพ์เล็ก
  • ใช้ camelCase สำหรับชื่อที่มีหลายคำ
  • เลือกชื่อที่สื่อความหมาย
function sayHello() {} // ดี
function calculateTotal() {} // ดี
function getUserAge() {} // ดี

การเรียกใช้ (รัน) ฟังก์ชัน

การสร้างฟังก์ชันไม่ได้ทำให้ฟังก์ชันทำงานโดยอัตโนมัติ หากต้องการให้ฟังก์ชันทำงาน คุณต้อง เรียกใช้ มัน:

function sayHi() {
  console.log("Hi")
}

sayHi() // บรรทัดนี้เรียกใช้ฟังก์ชัน

คุณสามารถเรียกใช้ฟังก์ชันเดิมได้หลายครั้ง:

sayHi() // แสดงผล "Hi"
sayHi() // แสดงผล "Hi" อีกครั้ง

ฟังก์ชันที่มีพารามิเตอร์

ฟังก์ชันจะทรงพลังยิ่งขึ้นเมื่อสามารถรับข้อมูลได้ ข้อมูลนำเข้าเหล่านี้เรียกว่า พารามิเตอร์:

function printName(name) {
  console.log(name)
}

printName("Kyle") // แสดงผล "Kyle"
printName("Sarah") // แสดงผล "Sarah"

พารามิเตอร์กับอาร์กิวเมนต์

  • พารามิเตอร์ คือชื่อที่ระบุไว้ในนิยามฟังก์ชัน - name
  • อาร์กิวเมนต์ คือค่าจริงที่ส่งเข้าไปเมื่อเรียกใช้ฟังก์ชัน - "Kyle", "Sarah"

ในทางเทคนิค คำทั้งสองมีความหมายต่างกัน แต่หลายคน (รวมถึงผมด้วย) ใช้คำเหล่านี้แทนกันในการสนทนาทั่วไป

พารามิเตอร์หลายตัว

คุณสามารถมีพารามิเตอร์หลายตัวโดยใช้เครื่องหมายจุลภาคคั่น:

function sum(a, b) {
  console.log(a + b)
}

sum(1, 2) // แสดงผล 3
sum(5, 7) // แสดงผล 12

การใช้ตัวแปรเป็นอาร์กิวเมนต์

คุณสามารถส่งตัวแปรเข้าไปในฟังก์ชันได้:

function add(x, y) {
  console.log(x + y)
}

let num1 = 10
let num2 = 5

add(num1, num2) // แสดงผล 15
add(3, 7) // คุณสามารถส่งค่าเข้าไปโดยตรงได้เช่นกัน

การส่งค่ากลับด้วย return

จนถึงตอนนี้ ฟังก์ชันของเราแสดงผลลัพธ์เพียงอย่างเดียว แต่ฟังก์ชันยังสามารถ ส่งค่ากลับ ด้วยคีย์เวิร์ด return ได้อีกด้วย ซึ่งช่วยให้คุณนำผลลัพธ์ไปใช้ที่อื่นในโค้ดได้:

function add(x, y) {
  return x + y
}

let sum = add(2, 3)
console.log(sum) // 5
console.log(add(10, 7)) // 17

เมื่อฟังก์ชันทำงานถึงคำสั่ง return ฟังก์ชันจะหยุดทำงานทันทีและส่งค่ากลับมา จากนั้นคุณสามารถเก็บค่านั้นไว้ในตัวแปรหรือนำไปใช้โดยตรงได้

หากคุณไม่ใช้ return ฟังก์ชันจะส่งค่า undefined กลับมาโดยค่าเริ่มต้น

ข้อผิดพลาดทางไวยากรณ์ที่พบบ่อย

  • ไม่มีวงเล็บ

    // ❌ ผิด - ไม่มีวงเล็บ
function sayHello {
  console.log("Hello")
}

// ✅ ถูกต้อง
function sayHello() {
  console.log("Hello")
}

  • ไม่มีวงเล็บปีกกา

    // ❌ ผิด - ไม่มีวงเล็บปีกกาเปิด
function sayHello()
  console.log("Hello")
}

// ❌ ผิด - ไม่มีวงเล็บปีกกาปิด
function sayHello() {
  console.log("Hello")

// ✅ ถูกต้อง
function sayHello() {
  console.log("Hello")
}

  • ลืมเรียกใช้ฟังก์ชัน

    // บรรทัดเหล่านี้สร้างฟังก์ชัน แต่ไม่ได้เรียกใช้
function sayHello() {
  console.log("Hello")
}

// คุณต้องเรียกใช้ฟังก์ชัน:
sayHello() // ตอนนี้ฟังก์ชันทำงานแล้ว

ประโยชน์ของฟังก์ชัน

  1. หลีกเลี่ยงการเขียนซ้ำ - เขียนตรรกะครั้งเดียว แล้วใช้ซ้ำได้หลายครั้ง
  2. แก้ไขได้ง่าย - เปลี่ยนตรรกะได้ในที่เดียว
  3. จัดระเบียบได้ดีขึ้น - แบ่งปัญหาที่ซับซ้อนออกเป็นส่วนย่อย ๆ
