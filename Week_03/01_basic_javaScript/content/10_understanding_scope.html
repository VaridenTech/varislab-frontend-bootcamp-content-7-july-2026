ทำความเข้าใจขอบเขตของตัวแปร (Scope)

ขอบเขตของตัวแปร (Scope) เป็นแนวคิดพื้นฐานที่สำคัญมากใน JavaScript ซึ่งทำหน้าที่กำหนดว่า ตัวแปรหรือ Function ที่เราประกาศขึ้นมานั้น จะสามารถเรียกใช้งานหรือเข้าถึงได้จากบริเวณใดของโค้ดบ้าง

function sayHi() {
  const result = "Hi"
  console.log(result) // แสดงผล "Hi"
}

const result = "Bye"
sayHi() // ผลลัพธ์บรรทัดนี้คืออะไร?
console.log(result) // ผลลัพธ์บรรทัดนี้คืออะไร?

Scope คืออะไร?

Scope คือขอบเขตพื้นที่ในโค้ดที่ตัวแปรหรือ Function นั้นมีตัวตนอยู่และสามารถเรียกใช้ได้ โดยทั่วไป JavaScript จะสร้างขอบเขต (Scope) ใหม่ขึ้นมาทุกครั้งที่มีการใช้วงเล็บปีกกา {} และจะมีขอบเขตใหญ่สุดที่เรียกว่า Global Scope ครอบคลุมโค้ดทั้งไฟล์

function myFunction() {
  // 👈 เริ่มต้น Scope ใหม่ภายใน Function
  const x = 5
}

// 👈 อยู่ใน Global Scope (นอก Function)
const x = 10
myFunction()

กฎการทำงานของ Scope: การเข้าถึงข้อมูลแบบทางเดียว (One-Way Street)

วิธีทำความเข้าใจการทำงานของ Scope ที่ง่ายที่สุดคือ ให้จินตนาการถึง "กระจกเงาด้านเดียว" (One-Way Mirror) ที่คนข้างในสามารถมองเห็นคนข้างนอกได้ แต่คนข้างนอกไม่สามารถมองเห็นคนข้างในได้:

  • ✅ Scope ด้านใน สามารถเข้าถึงตัวแปรใน Scope ด้านนอกได้

    const a = 1 // Global scope

function func() {
  const b = 2 // Function scope
  console.log(a) // ✅ มองเห็น scope ด้านนอกได้
  console.log(b) // ✅ มองเห็น scope ของตัวเองได้
}

  • ❌ Scope ด้านนอก ไม่สามารถเข้าถึงตัวแปรใน Scope ด้านในได้

    function func() {
  const b = 2 // Function scope
}

func()
console.log(b) // ❌ Error! ไม่สามารถเข้าถึงตัวแปร b ที่อยู่ข้างใน Function ได้

ประเภทของ Scope ใน JavaScript

ขอบเขตหรือ Scope ใน JavaScript สามารถแบ่งออกเป็น 3 ประเภทหลัก ๆ ที่คุณจำเป็นต้องรู้ ได้แก่ Global Scope, Function Scope และ Block Scope

1. Global Scope (ขอบเขตส่วนกลาง)

Global Scope คือขอบเขตที่อยู่นอกสุดของไฟล์โค้ด ตัวแปรหรือ Function ใด ๆ ที่ประกาศไว้ในพื้นที่นี้จะถือเป็น "ตัวแปรส่วนกลาง" และสามารถถูกเรียกใช้งานจากส่วนใดของโค้ดก็ได้ (รวมถึงภายใน Function ด้วย):

const name = "Kyle" // Global scope

function sayHi() {
  console.log(name) // ✅ สามารถเข้าถึงตัวแปร name จาก Global Scope ได้
}

sayHi()

2. Block Scope (ขอบเขตแบบบล็อก)

Block Scope คือขอบเขตที่จะถูกสร้างขึ้นใหม่ทุกครั้งเมื่อมีวงเล็บปีกกา {} เช่น ในคำสั่งเงื่อนไข if, การวนลูป for / while หรือแม้กระทั่งการเขียนปีกกาเปล่า ๆ ครอบโค้ดไว้:

function myFunction() {
  const x = 3 // Block scope ใหม่
}

if (true) {
  const x = 1 // Block scope ใหม่
}

{
  const x = 2 // Block scope ใหม่
}

3. Function Scope (ขอบเขตแบบ Function)

Function Scope คือขอบเขตที่จะถูกสร้างขึ้นใหม่ทุกครั้งที่มีการประกาศ Function ซึ่งในทางปฏิบัติจะคล้ายกับ Block Scope มาก แต่มีความสำคัญเป็นพิเศษเมื่อทำงานร่วมกับ Keyword รุ่นเก่าอย่าง var (ซึ่งจะอธิบายเพิ่มเติมในภายหลัง):

function myFunction() {
  const x = 5
}

ขอบเขตที่ซ้อนกัน (Nested Scopes)

เราสามารถเขียนบล็อกหรือ Function ซ้อนกันหลาย ๆ ชั้นได้ ซึ่งจะทำให้เกิด Scope ซ้อนกันเป็นลำดับขั้น:

let c = 3 // Global scope

{
  // Block scope ด้านนอก
  let a = 1

  {
    // Block scope ด้านใน
    let b = 2

    console.log(a) // 1 - มองเห็น scope ด้านนอกได้
    console.log(b) // 2 - มองเห็น scope ของตัวเองได้
    console.log(c) // 3 - มองเห็น global scope ได้
  }

  console.log(a) // 1 - มองเห็น scope ของตัวเองได้
  console.log(c) // 3 - มองเห็น global scope ได้
  // console.log(b)  // ❌ ข้อผิดพลาด - มองไม่เห็น scope ด้านใน
}

การประกาศตัวแปรชื่อซ้ำกัน (Variable Shadowing)

เราสามารถตั้งชื่อตัวแปรซ้ำกันในคนละ Scope ได้โดยไม่ทำให้โปรแกรมพังหรือเกิดข้อผิดพลาด:

const result = "Kyle" // Global scope

function sayHi(name) {
  const result = "Hi " + name // Block scope - เป็นตัวแปรคนละตัว!
  console.log(result) // "Hi Kyle"
}

sayHi("Kyle")
console.log(result) // "Kyle" - ยังคงเป็นตัวแปรส่วนกลาง

ตัวแปรทั้งสองจะถูกมองเป็นคนละกล่องข้อมูลแยกจากกันอย่างสิ้นเชิง เพียงแค่ใช้ชื่อเดียวกันเท่านั้น

ลำดับการค้นหาตัวแปรของ JavaScript

เมื่อมีการเรียกใช้ตัวแปร JavaScript Engine จะเริ่มค้นหาตัวแปรนั้นจาก Scope ปัจจุบันก่อนเสมอ หากไม่พบ จึงจะค่อย ๆ ค้นหาไล่ออกไปยัง Scope ชั้นนอกทีละระดับจนถึง Global Scope:

const a = 1 // Global scope
const name = "Kyle" // Global scope

function myFunction() {
  const a = 2 // Block scope

  console.log(a) // แสดงผล 2
  console.log(name) // แสดงผล "Kyle"
}

myFunction()

แนวทางปฏิบัติที่ดี (Best Practices)

  • 1. หลีกเลี่ยงการใช้ตัวแปร Global โดยไม่จำเป็น

    // ❌ ไม่แนะนำ: ประกาศตัวแปรไว้ใน Global Scope ทั้งที่ใช้แค่ใน Function เดียว
const name = "Kyle"

function myFunction() {
  console.log(name)
}
    // ✅ แนะนำ: ประกาศตัวแปรไว้ภายในขอบเขตที่ต้องการใช้งานจริง ๆ
function myFunction() {
  const name = "Kyle"
  console.log(name)
}

  • 2. หลีกเลี่ยงการตั้งชื่อตัวแปรซ้ำกันใน Scope ที่ซ้อนกัน

    // ❌ ไม่แนะนำ: ตั้งชื่อซ้ำกันในบล็อกที่ซ้อนกัน ทำให้โค้ดอ่านยากและสับสน
const a = 1

function myFunction() {
  const a = 2
  console.log(a) // นี่คือ 'a' ตัวใด?
}
    // ✅ แนะนำ: ใช้ชื่อที่แตกต่างและสื่อความหมายชัดเจน
const outerA = 1

function myFunction() {
  const innerA = 2
  console.log(innerA) // เรียกใช้ตัวแปรได้อย่างมั่นใจและเข้าใจง่าย
}
