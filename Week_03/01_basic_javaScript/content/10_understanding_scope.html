ทำความเข้าใจ Scope

Scope เป็นแนวคิดพื้นฐานใน JavaScript ที่กำหนดว่าสามารถเข้าถึงตัวแปรและฟังก์ชันได้จากที่ใด

function sayHi() {
  const result = "Hi"
  console.log(result) // แสดงผล "Hi"
}

const result = "Bye"
sayHi() // บรรทัดนี้แสดงผลอะไร?
console.log(result) // บรรทัดนี้แสดงผลอะไร?

Scope คืออะไร?

Scope คือบริเวณในโค้ดที่สามารถเข้าถึงตัวแปรหรือฟังก์ชันได้ JavaScript จะสร้าง scope ใหม่ภายในวงเล็บปีกกา {} ทุกชุด และยังมี global scope สำหรับทั้งไฟล์ด้วย

function myFunction() {
  // สร้าง scope ใหม่ตรงนี้
  const x = 5
}

// Global scope
const x = 10
myFunction()

Scope ทำงานทางเดียว

วิธีคิดเกี่ยวกับ scope ที่ดีที่สุดคือมองว่าเป็น กระจกมองทางเดียว ซึ่งมองออกไปด้านนอกเสมอ:

  • ✅ Scope ด้านในมองเห็น scope ด้านนอกได้

    const a = 1 // Global scope

function func() {
  const b = 2 // Function scope
  console.log(a) // ✅ มองเห็น scope ด้านนอกได้
  console.log(b) // ✅ มองเห็น scope ของตัวเองได้
}

  • ❌ Scope ด้านนอกมองไม่เห็น scope ด้านใน

    function func() {
  const b = 2 // Function scope
}

func()
console.log(b) // ❌ ข้อผิดพลาด - มองไม่เห็น scope ด้านใน

ประเภทของ Scope

Scope ใน JavaScript มีหลายประเภท แต่สามประเภทที่สำคัญที่สุดคือ global scope, function scope และ block scope

Global Scope

Global scope คือ scope ชั้นนอกสุด ซึ่งก็คือทั้งไฟล์ของคุณ ตัวแปรที่ประกาศไว้ตรงนี้สามารถเข้าถึงได้จากทุกที่ในโค้ด:

const name = "Kyle" // Global scope

function sayHi() {
  console.log(name) // ✅ เข้าถึงตัวแปรส่วนกลางได้
}

sayHi()

Block Scope

Block scope ใหม่จะถูกสร้างขึ้นสำหรับโค้ดใด ๆ ที่อยู่ระหว่างวงเล็บปีกกา {}:

function myFunction() {
  const x = 3 // Block scope ใหม่
}

if (true) {
  const x = 1 // Block scope ใหม่
}

{
  const x = 2 // Block scope ใหม่
}

Function Scope

Function scope จะถูกสร้างขึ้นทุกครั้งที่คุณกำหนดฟังก์ชัน Function scope คล้ายกับ block scope มาก แต่คุณสมบัติบางอย่างของ JavaScript (ซึ่งจะกล่าวถึงในภายหลัง) ใช้ function scope แทน block scope:

function myFunction() {
  const x = 5
}

Scope ที่ซ้อนกัน

คุณสามารถมี scope ได้หลายระดับ:

let c = 3 // Global scope

{
  // Block scope ด้านนอก
  let a = 1

  {
    // Block scope ด้านใน
    let b = 2

    console.log(a) // 1 - มองเห็น scope ด้านนอกได้
    console.log(b) // 2 - มองเห็น scope ของตัวเองได้
    console.log(c) // 3 - มองเห็น global scope ได้
  }

  console.log(a) // 1 - มองเห็น scope ของตัวเองได้
  console.log(c) // 3 - มองเห็น global scope ได้
  // console.log(b)  // ❌ ข้อผิดพลาด - มองไม่เห็น scope ด้านใน
}

ชื่อที่ขัดแย้งกัน

คุณสามารถมีตัวแปรชื่อเดียวกันใน scope ที่ต่างกันได้โดยไม่เกิดความขัดแย้ง:

const result = "Kyle" // Global scope

function sayHi(name) {
  const result = "Hi " + name // Block scope - เป็นตัวแปรคนละตัว!
  console.log(result) // "Hi Kyle"
}

sayHi("Kyle")
console.log(result) // "Kyle" - ยังคงเป็นตัวแปรส่วนกลาง

ตัวแปรเหล่านี้แยกจากกันอย่างสิ้นเชิง เพียงแต่บังเอิญใช้ชื่อเดียวกัน

การพิจารณาว่าจะใช้ตัวแปรใด

เมื่อ JavaScript มองหาตัวแปร มันจะเริ่มจาก scope ปัจจุบันแล้วไล่ออกไปด้านนอก:

const a = 1 // Global scope
const name = "Kyle" // Global scope

function myFunction() {
  const a = 2 // Block scope

  console.log(a) // แสดงผล 2
  console.log(name) // แสดงผล "Kyle"
}

myFunction()

แนวทางปฏิบัติที่ดี

  • ลดการใช้ตัวแปรส่วนกลางให้น้อยที่สุด

    // ❌ หลีกเลี่ยงการทำแบบนี้
const name = "Kyle"

function myFunction() {
  console.log(name)
}
    // ✅ วางตัวแปรไว้ภายในมากที่สุดเท่าที่ทำได้
function myFunction() {
  const name = "Kyle"
  console.log(name)
}

  • อย่าใช้ชื่อเดียวกันสำหรับตัวแปรคนละตัวใน scope ที่ซ้อนกัน

    // ❌ ชวนให้สับสน - ยากที่จะบอกว่าใช้ 'a' ตัวใด
const a = 1

function myFunction() {
  const a = 2
  console.log(a) // นี่คือ 'a' ตัวใด?
}
    // ✅ ชัดเจน - ใช้ชื่อต่างกัน
const outerA = 1

function myFunction() {
  const innerA = 2
  console.log(innerA) // ใช้ตัวแปรใน scope ด้านในอย่างชัดเจน
}
