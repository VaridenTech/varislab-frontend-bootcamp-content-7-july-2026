การส่งฟังก์ชันเป็นอาร์กิวเมนต์

หนึ่งในคุณสมบัติที่ทรงพลังที่สุดของ JavaScript คือความสามารถในการส่งฟังก์ชันไปยังฟังก์ชันอื่นในรูปแบบอาร์กิวเมนต์

⚠️ คำเตือน: นี่เป็นหัวข้อที่เข้าใจยาก ดังนั้นค่อย ๆ ใช้เวลาทำความเข้าใจให้ถ่องแท้!

ฟังก์ชันคือตัวแปร

สิ่งสำคัญที่ต้องเข้าใจคือ ฟังก์ชันเป็นเพียงตัวแปร ที่เก็บตรรกะ:

function printName(name) {
  console.log(name)
}

// จะเกิดอะไรขึ้นหากเราแสดงชื่อฟังก์ชันโดยไม่มีวงเล็บ?
console.log(printName) // แสดงผล: ƒ printName(name) { console.log(name) }

เมื่อคุณสร้างฟังก์ชันด้วยคีย์เวิร์ด function JavaScript จะสร้างตัวแปรชื่อนั้นและเก็บนิยามฟังก์ชันไว้ในตัวแปร เช่นเดียวกับที่ let สร้างตัวแปรเพื่อเก็บค่า

การส่งฟังก์ชันเบื้องต้น

เนื่องจากฟังก์ชันเป็นตัวแปร คุณจึงส่งฟังก์ชันไปยังฟังก์ชันอื่นได้:

function printName(name) {
  console.log(name)
}

function callFunction(x) {
  console.log("before")
  x("Kyle") // เรียกใช้ฟังก์ชันที่ถูกส่งเข้ามา
  console.log("after")
}

// ส่งฟังก์ชันเข้าไป (สังเกตว่าไม่มีวงเล็บ!)
callFunction(printName)
// ผลลัพธ์:
// before
// Kyle
// after

ประเด็นสำคัญ: ฟังก์ชันกับการเรียกใช้ฟังก์ชัน

  • printVariable - นี่คือตัวฟังก์ชันเอง (ตัวแปร)
  • printVariable() - นี่คือการเรียกใช้ฟังก์ชันและรันโค้ดที่อยู่ข้างใน

ฟังก์ชัน Callback

เมื่อคุณส่งฟังก์ชันไปยังฟังก์ชันอื่น ฟังก์ชันที่ถูกส่งเข้าไปนั้นเรียกว่า callback:

function sumCallback(a, b, callback) {
  let sum = a + b
  callback(sum) // เรียกใช้ callback พร้อมส่งผลลัพธ์เข้าไป
}

function handleSum(sum) {
  console.log(sum)
}

sumCallback(1, 2, handleSum) // แสดงผล: 3

นี่คือสิ่งที่เกิดขึ้นทีละขั้นตอน:

  1. sumCallback รับ 1, 2 และฟังก์ชัน handleSum
  2. ฟังก์ชันคำนวณ 1 + 2 = 3
  3. ฟังก์ชันเรียกใช้ callback (handleSum) พร้อมผลลัพธ์ 3
  4. handleSum แสดงค่า sum

ข้อผิดพลาดที่พบบ่อย: เรียกใช้แทนที่จะส่งฟังก์ชัน

ข้อผิดพลาดที่พบบ่อยมากคือการเรียกใช้ฟังก์ชันโดยไม่ตั้งใจ แทนที่จะส่งตัวฟังก์ชันเข้าไป:

function printName(name) {
  console.log(name)
}

function printGreeting(name, callback) {
  callback("Hello " + name)
}

// ❌ ผิด - บรรทัดนี้เรียกใช้ printName ทันที
printGreeting("Kyle", printName())
// ผลลัพธ์: undefined ถูกส่งเข้าไป จากนั้นเกิดข้อผิดพลาด "callback is not a function"

// ✅ ถูกต้อง - บรรทัดนี้ส่งตัวฟังก์ชันเข้าไป
printGreeting("Kyle", printName)
// ผลลัพธ์: "Hello Kyle"

เมื่อคุณเติมวงเล็บ คุณกำลังเรียกใช้ฟังก์ชันและส่งค่าที่ฟังก์ชันนั้นส่งกลับมาเข้าไป (ซึ่งจะเป็น undefined หากฟังก์ชันไม่ได้ส่งค่าใดกลับมา)

ตัวอย่างกรณีใช้งาน

function doMath(a, b, operation) {
  return operation(a, b)
}

function add(x, y) {
  return x + y
}

function subtract(x, y) {
  return x - y
}

console.log(doMath(5, 3, add)) // แสดงผล: 8
console.log(doMath(5, 3, subtract)) // แสดงผล: 2

แบบฝึกหัด

ลองสร้างฟังก์ชันที่:

  1. รับพารามิเตอร์สามตัว ได้แก่ firstName, lastName และฟังก์ชัน callback
  2. สร้างตัวแปร fullName โดยนำชื่อและนามสกุลมาต่อกัน
  3. ส่ง fullName ไปยังฟังก์ชัน callback (ซึ่งเติม "Hello " ไว้ข้างหน้า)
  4. แสดงค่าที่ฟังก์ชัน callback ส่งกลับมา
เฉลย 
function printGreeting(firstName, lastName, callback) {
  const fullName = firstName + " " + lastName
  console.log(callback(fullName))
}

function getGreeting(name) {
  return "Hello " + name
}

printGreeting("Kyle", "Cook", getGreeting) // "Hello Kyle Cook"

ทำไมจึงใช้ Callback?

  1. ความยืดหยุ่น - ฟังก์ชันเดียวกันสามารถทำงานต่างกันได้ตาม callback
  2. การแยกหน้าที่รับผิดชอบ - ฟังก์ชันหลักจัดการเรื่องหนึ่ง ส่วน callback จัดการอีกเรื่องหนึ่ง
