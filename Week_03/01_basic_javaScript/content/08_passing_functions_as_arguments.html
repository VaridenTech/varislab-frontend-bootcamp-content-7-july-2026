การส่ง Function เป็น Argument (Passing Functions as Arguments)

หนึ่งในฟีเจอร์ที่ทรงพลังและยืดหยุ่นที่สุดของภาษา JavaScript คือความสามารถในการส่ง Function หนึ่ง เข้าไปเป็น Argument (Argument) ของอีก Function หนึ่งได้

💡 คำแนะนำ: หัวข้อนี้อาจจะทำความเข้าใจได้ยากสำหรับผู้เริ่มต้น แนะนำให้ค่อย ๆ อ่านและลองเขียนโค้ดตามไปทีละขั้นตอนนะครับ

Function ก็คือตัวแปรประเภทหนึ่ง (First-Class Functions)

แนวคิดสำคัญที่คุณต้องเข้าใจก่อนคือ ใน JavaScript Function ก็เปรียบเสมือนตัวแปรทั่วไป ที่เก็บชุดคำสั่งหรือตรรกะการทำงานเอาไว้:

function printName(name) {
  console.log(name)
}

// จะเกิดอะไรขึ้นหากเราแสดงชื่อ Function โดยไม่มีวงเล็บ?
console.log(printName) // แสดงผล: ƒ printName(name) { console.log(name) }

เมื่อเราสร้าง Function ขึ้นมา JavaScript Engine จะทำการสร้างตัวแปรชื่อเดียวกับ Function นั้น และเก็บนิยามหรือโค้ดของ Function เอาไว้ข้างใน ซึ่งมีลักษณะการทำงานคล้ายกับการใช้ Keyword let ประกาศตัวแปรเพื่อเก็บข้อมูลทั่วไป

การส่ง Function เข้าไปทำงานในอีก Function หนึ่ง

เนื่องจาก Function มีสถานะเป็นตัวแปรประเภทหนึ่ง เราจึงสามารถส่งมันเข้าไปเป็น Argument ของ Function อื่นได้เหมือนกับข้อมูลทั่วไป:

function printName(name) {
  console.log(name)
}

function callFunction(x) {
  console.log("before")
  x("Kyle") // เรียกใช้ Function ที่ถูกส่งเข้ามา
  console.log("after")
}

// ส่งตัวแปร Function เข้าไปเป็น Argument (สังเกตว่าไม่มีวงเล็บ () ต่อท้าย!)
callFunction(printName)
// ผลลัพธ์:
// before
// Kyle
// after

ความแตกต่างสำคัญ: Function vs การเรียกใช้งาน Function

  • printName - หมายถึง ตัวแปรที่เก็บ Function นั้นไว้ (ส่งไปเพื่อเรียกใช้ทีหลัง)
  • printName() - หมายถึง การสั่งให้ Function นั้นทำงานทันที ณ บรรทัดนั้น

Function คอลแบ็ก (Callback Functions)

Function ที่เราส่งเข้าไปเป็น Argument เพื่อให้ Function หลักเรียกใช้งานในภายหลังนั้น เราจะเรียกว่า Function คอลแบ็ก (Callback Function):

function sumCallback(a, b, callback) {
  let sum = a + b
  callback(sum) // เรียกใช้งาน Function callback พร้อมส่งผลลัพธ์กลับไปประมวลผล
}

function handleSum(sum) {
  console.log(sum)
}

sumCallback(1, 2, handleSum) // แสดงผล: 3

กระบวนการทำงานทีละขั้นตอน:

  1. sumCallback รับค่า 1, 2 และตัวแปร Function handleSum เข้ามา
  2. คำนวณผลรวมได้ 1 + 2 = 3
  3. เรียกใช้งาน Function callback (ซึ่งก็คือ handleSum) พร้อมส่งผลลัพธ์ 3 เข้าไป
  4. Function handleSum ทำงานและแสดงผลลัพธ์ 3 ออกมา

ข้อผิดพลาดที่พบบ่อย: การเผลอเรียกใช้ Function ทันที

ข้อผิดพลาดที่พบบ่อยที่สุดสำหรับผู้เริ่มต้นคือ การเผลอใส่เครื่องหมายวงเล็บ () ซึ่งเป็นการสั่งให้ Function ทำงานทันที แทนที่จะส่งเฉพาะตัวแปร Function เข้าไป:

function printName(name) {
  console.log(name)
}

function printGreeting(name, callback) {
  callback("Hello " + name)
}

// ❌ ผิด - บรรทัดนี้จะสั่งให้ printName ทำงานทันที
printGreeting("Kyle", printName())
// ผลลัพธ์: ค่าที่ส่งกลับ (ซึ่งคือ undefined) จะถูกส่งเข้าไปแทนตัว Function และทำให้เกิดข้อผิดพลาด "callback is not a function"

// ✅ ถูกต้อง - ส่งเฉพาะชื่อตัวแปร Function เข้าไป เพื่อให้เรียกใช้ทีหลัง
printGreeting("Kyle", printName)
// ผลลัพธ์: "Hello Kyle"

จำไว้ว่า: เมื่อใดก็ตามที่คุณใส่เครื่องหมายวงเล็บ () ต่อท้ายชื่อ Function จะหมายถึงคุณกำลังสั่งให้มันทำงานทันที และค่าที่จะถูกส่งต่อเข้าไปคือผลลัพธ์จากการ return ของ Function นั้น (ซึ่งจะเป็น undefined หากไม่มีการ return ค่าใด ๆ)

ตัวอย่างการใช้งานจริง

function doMath(a, b, operation) {
  return operation(a, b)
}

function add(x, y) {
  return x + y
}

function subtract(x, y) {
  return x - y
}

console.log(doMath(5, 3, add)) // แสดงผล: 8
console.log(doMath(5, 3, subtract)) // แสดงผล: 2

ลองทำแบบฝึกหัด

ฝึกฝนความเข้าใจด้วยการสร้าง Function ตามเงื่อนไขต่อไปนี้:

  1. รับ Parameter สามตัว ได้แก่ firstName, lastName และ Function callback
  2. สร้างตัวแปร fullName โดยนำชื่อและนามสกุลมาต่อกัน
  3. ส่งค่า fullName ไปยัง Function callback (ซึ่งจะทำหน้าที่เติมคำว่า "Hello " ไว้ข้างหน้า)
  4. แสดงผลลัพธ์ที่ได้รับกลับมาจาก Function callback
ดูเฉลย 
function printGreeting(firstName, lastName, callback) {
  const fullName = firstName + " " + lastName
  console.log(callback(fullName))
}

function getGreeting(name) {
  return "Hello " + name
}

printGreeting("Kyle", "Cook", getGreeting) // "Hello Kyle Cook"

ทำไมเราจึงต้องใช้ Callback?

  1. เพิ่มความยืดหยุ่น (Flexibility) - Function หลักตัวเดียวสามารถทำงานได้หลากหลายรูปแบบ ขึ้นอยู่กับพฤติกรรมของ Function callback ที่เราส่งเข้าไป
  2. การแยกหน้าที่รับผิดชอบ (Separation of Concerns) - ช่วยแบ่งหน้าที่การทำงานอย่างชัดเจน เช่น Function หลักทำหน้าที่คำนวณหาผลลัพธ์ ส่วน callback ทำหน้าที่นำผลลัพธ์นั้นไปแสดงผลในรูปแบบที่ต้องการ
