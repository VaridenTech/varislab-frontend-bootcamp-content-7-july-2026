Arrow Function

Arrow function เป็นอีกวิธีหนึ่งในการเขียนฟังก์ชันใน JavaScript โดยมีไวยากรณ์ที่สั้นกว่าและมีคุณสมบัติเฉพาะบางอย่างที่ทำให้มีประโยชน์อย่างยิ่งในบางสถานการณ์

การแปลงฟังก์ชันปกติเป็น Arrow Function

มาเริ่มด้วยฟังก์ชันปกติแล้วแปลงเป็น arrow function:

// ฟังก์ชันปกติ
function sum(a, b) {
  return a + b
}

// Arrow function
const sumArrow = (a, b) => {
  return a + b
}

console.log(sum(1, 2)) // 3
console.log(sumArrow(1, 2)) // 3

ไวยากรณ์ของ Arrow Function

const functionName = (parameters) => {
  // ส่วนคำสั่งของฟังก์ชัน
}

ความแตกต่างสำคัญจากฟังก์ชันปกติ:

  1. ใช้ const (หรือ let) แทนคีย์เวิร์ด function
  2. ชื่อฟังก์ชันอยู่ก่อน ตามด้วย =
  3. พารามิเตอร์อยู่ในวงเล็บ
  4. มีลูกศร (=>) คั่นระหว่างพารามิเตอร์กับส่วนคำสั่ง
  5. ส่วนคำสั่งของฟังก์ชันอยู่ในวงเล็บปีกกา

คุณสมบัติเฉพาะ

โดยส่วนใหญ่ คุณสามารถใช้ arrow function ได้เหมือนฟังก์ชันปกติ แต่มีคุณสมบัติเฉพาะบางอย่างที่คุณต้องทราบ

รูปแบบย่อสำหรับพารามิเตอร์เดียว

เมื่อมี พารามิเตอร์เพียงหนึ่งตัว คุณสามารถละวงเล็บได้:

// มีวงเล็บ
const printName = (name) => {
  console.log(name)
}

// ไม่มีวงเล็บ
const printNameShort = name => {
  console.log(name)
}

หมายเหตุ: หากไม่มีพารามิเตอร์หรือมีพารามิเตอร์หลายตัว จำเป็นต้องใช้วงเล็บ:

// ไม่มีพารามิเตอร์ - ต้องมีวงเล็บ
const sayHi = () => {
  console.log("Hi")
}

// พารามิเตอร์หลายตัว - ต้องมีวงเล็บ
const add = (x, y) => {
  return x + y
}

รูปแบบย่อสำหรับการส่งค่ากลับในบรรทัดเดียว

เมื่อส่วนคำสั่งของฟังก์ชันมีเพียง หนึ่งบรรทัดที่ส่งค่ากลับ คุณสามารถใช้ไวยากรณ์ที่สั้นยิ่งขึ้นได้:

// Arrow function แบบปกติ
const sum = (a, b) => {
  return a + b
}

// Arrow function แบบย่อ (ส่งค่ากลับโดยปริยาย)
const sumShort = (a, b) => a + b

console.log(sumShort(1, 2)) // 3

JavaScript จะส่งค่าหลังลูกศรกลับโดยอัตโนมัติโดยไม่ต้องใช้ return หรือวงเล็บปีกกา หากมีเพียงบรรทัดเดียว

ข้อผิดพลาดที่พบบ่อย

  1. การส่งค่ากลับโดยปริยายไม่ถูกต้อง

    // ❌ ผิด - ใช้ 'return' โดยไม่มีวงเล็บปีกกาไม่ได้
const add = (a, b) => return a + b

// ❌ ผิด - ไม่ได้ส่งค่าใดกลับมา
const add = (a, b) => {
  a + b
}

// ✅ ถูกต้อง - ใช้วงเล็บปีกกาพร้อม return
const add = (a, b) => {
  return a + b
}

// ✅ หรือใช้การส่งค่ากลับโดยปริยายโดยไม่มีวงเล็บปีกกา
const add = (a, b) => a + b

  2. ลืมวงเล็บเมื่อมีพารามิเตอร์หลายตัว

    // ❌ ผิด - พารามิเตอร์หลายตัวต้องมีวงเล็บ
const multiply = a, b => a * b

// ✅ ถูกต้อง
const multiply = (a, b) => a * b

เมื่อใดควรใช้ Arrow Function

function processData(x, callback) {
  callback(x)
}

// ใช้ฟังก์ชันปกติ (ยืดยาว)
processData(10, function (variable) {
  console.log(variable)
})

// ใช้ arrow function (กระชับ)
processData(10, (variable) => {
  console.log(variable)
})

// หรือสั้นกว่านั้นด้วยการเขียนในบรรทัดเดียว
processData(10, (variable) => console.log(variable))

ใช้ arrow function สำหรับ:

  • ฟังก์ชันสั้น ๆ ที่ไม่ซับซ้อน
  • ฟังก์ชัน callback
  • ฟังก์ชันนิรนาม
  • ฟังก์ชันที่เขียนได้ในบรรทัดเดียว

พิจารณาใช้ฟังก์ชันปกติสำหรับ:

  • ฟังก์ชันที่ต้องการชื่อซึ่งสื่อความหมาย
