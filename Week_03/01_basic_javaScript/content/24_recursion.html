Recursion

Recursion คือการที่ฟังก์ชันเรียกตัวเอง แม้ฟังดูแปลกในตอนแรก แต่เป็นเทคนิคที่ทรงพลังสำหรับแก้ปัญหาที่แบ่งเป็นปัญหาย่อยคล้ายกันได้

⚠️ คำเตือน: หัวข้อนี้เข้าใจยาก จึงควรใช้เวลาทำความเข้าใจให้ถ่องแท้!

ตัวอย่าง Recursion พื้นฐาน

function printNumbers(number) {
  if (number > 10) return // Base case: หยุด Recursion
  console.log(number)
  printNumbers(number + 1) // Recursive call: ฟังก์ชันเรียกตัวเอง
}
printNumbers(1) // แสดง 1 ถึง 10

ใช้แทน For loop แบบนี้ได้:

for (let i = 1; i <= 10; i++) console.log(i)

สองส่วนสำคัญของ Recursion

1. Base Case (เงื่อนไขหยุด)

เงื่อนไขที่หยุดไม่ให้ Recursion ทำงานตลอดไป:

if (number > 10) return

หากไม่มี Base case ที่เหมาะสม จะเกิด Recursion ไม่สิ้นสุดและ Stack overflow error:

// ❌ ทำให้โปรแกรมล่ม!
function printHi() {
  console.log("Hi")
  printHi() // RangeError: Maximum call stack size exceeded
}
printHi()

2. Recursive Case (การเรียกตัวเอง)

ฟังก์ชันเรียกตัวเองพร้อมพารามิเตอร์ที่เปลี่ยนไป:

printNumbers(number + 1) // เรียกตัวเองด้วยเลขถัดไป

ทำความเข้าใจ Call Stack

เมื่อฟังก์ชันเรียกตัวเอง แต่ละครั้งจะถูกเพิ่มลงใน Call stack:

function countdown(n) {
  console.log(n)
  if (n <= 0) return
  console.log("Before")
  countdown(n - 1)
  console.log("After")
}
countdown(3)

ภาพ Call stack:

countdown(3) เรียก countdown(2)
  countdown(2) เรียก countdown(1)
    countdown(1) เรียก countdown(0)
      countdown(0) ส่งค่ากลับ
    countdown(1) ส่งค่ากลับ
  countdown(2) ส่งค่ากลับ
countdown(3) ส่งค่ากลับ

เมื่อใดควรใช้ Recursion

โครงสร้างข้อมูลแบบต้นไม้:

const fileSystem = {
  name: "root", type: "folder",
  children: [
    { name: "documents", type: "folder", children: [
      { name: "resume.pdf", type: "file" },
      { name: "notes.txt", type: "file" },
    ]},
    { name: "photos", type: "folder", children: [
      { name: "vacation.jpg", type: "file" },
    ]},
    { name: "readme.txt", type: "file" },
  ],
}

function countFiles(item) {
  if (item.type === "file") return 1 // Base case
  let count = 0
  for (const child of item.children) {
    count += countFiles(child) // เรียกซ้ำสำหรับลูกแต่ละตัว
  }
  return count
}
console.log(countFiles(fileSystem)) // ไฟล์ทั้งหมด 4 ไฟล์

ข้อผิดพลาดที่พบบ่อย

  1. ลืม Base Case

    // ✅ มี Base case
function countdown(n) {
  if (n <= 0) return
  console.log(n)
  countdown(n - 1)
}

// ❌ ไม่มี Base case
function badCountdown(n) {
  console.log(n)
  badCountdown(n - 1)
}

  2. ไม่เคลื่อนไปหา Base Case

    // ✅ n ลดลงทุกครั้ง
function countdown(n) {
  if (n <= 0) return
  console.log(n)
  countdown(n - 1)
}

// ❌ n ไม่เปลี่ยน
function badCountdown(n) {
  if (n <= 0) return
  console.log(n)
  countdown(n)
}

เมื่อใดควรใช้ Recursion

✅ ใช้เมื่อ:

  • ทำงานกับโครงสร้างแบบต้นไม้หรือข้อมูลซ้อนกัน
  • แบ่งปัญหาเป็นปัญหาย่อยคล้ายกันได้
  • วิธี Recursive ชัดเจนกว่าวิธีวนซ้ำ
  • ประมวลผลข้อมูลลำดับชั้น

❌ หลีกเลี่ยงเมื่อ:

  • Loop ธรรมดาทำงานได้ดีเท่ากัน
  • ประสิทธิภาพสำคัญมาก
  • วิธี Recursive เข้าใจยาก

แบบฝึกหัด

เขียนฟังก์ชัน Recursive เพื่อหาค่าสูงสุดใน Array ซ้อนกัน:

const data = [1, [2, 3], [4, [5, 6]], 7]
// ควรส่งกลับ 7
เฉลย
function findMax(arr) {
  let max = -Infinity
  for (const item of arr) {
    if (Array.isArray(item)) {
      const subMax = findMax(item)
      max = Math.max(max, subMax)
    } else {
      max = Math.max(max, item)
    }
  }
  return max
}
const data = [1, [2, 3], [4, [5, 6]], 7]
console.log(findMax(data)) // 7
